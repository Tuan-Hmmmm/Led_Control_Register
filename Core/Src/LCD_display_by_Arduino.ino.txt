void setup()
{
  lcd.begin(16,2);
  lcd.init();
  lcd.backlight();
  Serial.begin(9600);
  mySerial.begin(9600);
  lcd.setCursor(1,0);
  lcd.print("LED RED   : ");
  lcd.setCursor(1,1);
  lcd.print("LED YELLOW: ");
  Serial.println("Tuan");
}
void loop()
{
  if (mySerial.available())
  { 
    temp = (char)mySerial.read();
    Serial.print(temp);
    str[i]=temp;
    i++;
    if (i==6)
    {
       i=0;
       lcd.setCursor(12,0);
       lcd.print(str[0]);

       lcd.print(str[3]);
       lcd.print(str[4]);

    }
  }
}
